    public int storageHeight;     // 装备后展开网格行数
    public float armorValue;     // 护甲值

    // 行为列表（替代无限加字段）
    public List<ItemBehaviourEntry> behaviours;

    // 使用
    public float useTime;          // 使用耗时（秒）

    // 生存系统

    public float warmthValue;            // 保暖值
    public bool isWaterproof;
    public string foodType;       // 食物类型（Meat/Vegetable/Drink/Medicine）

    // 武器属性

#### PlacedItem（背包中已放置物品）
```csharp
[Serializable]
public struct PlacedItem
{
    public int instanceId;          // 唯一实例 ID（存档系统用）
    public ItemData itemData;       // 物品数据
    public int count;               // 堆叠数量
    public int gridX;               // 列坐标
    public int gridY;               // 行坐标
    public bool rotated;            // 是否已旋转 90°
    public bool isGhost;            // 是否为腰带武器槽幽灵占位格
    public EquipSlot ghostSourceSlot; // 幽灵格来源武器槽

    // 计算属性
    public int GridWidth => isGhost ? 1 :
        (itemData != null ? (rotated ? itemData.gridHeight : itemData.gridWidth) : 1);
    public int GridHeight => isGhost ? 1 :
        (itemData != null ? (rotated ? itemData.gridWidth : itemData.gridHeight) : 1);
    public float TotalWeight => itemData != null ? itemData.weight * count : 0f;
}
```

#### 3. ItemUsageSystem — 使用物品桥接

```csharp
public class ItemUsageSystem : MonoBehaviour
{
    private Inventory _inventory;
    private SurvivalSystem _survival;

    private void Start()
    {
        _inventory = ServiceLocator.Get<Inventory>();
        _survival = ServiceLocator.Get<SurvivalSystem>();
        EventBus.Subscribe<ItemUsedEvent>(OnItemUsed);
    }

    private void OnItemUsed(ItemUsedEvent evt)
    {
        // 1. 从背包扣除物品
        _inventory.RemoveItem(evt.ItemData, evt.Count);
        // 2. 遍历 itemEffects 列表，对每个效果调用 SurvivalSystem.ApplyItemEffect(eff)
        foreach (var eff in evt.ItemData.itemEffects)
            _survival.ApplyItemEffect(eff);
    }
}
```

### 模块间关联

```
                        配置层 (Config)
    ItemData / ItemCategory / ItemQuality / ItemUsageTag
    ItemEffect / EquipSlot / ItemGraph / ItemBehaviourEntry
                |                          |
                | 引用                      | 引用
                v                          v
    Inventory 背包系统              SurvivalSystem 生存系统
    - Inventory.cs                - Subscribe: EquipmentChanged
    - InventoryContainer.cs       - Subscribe: HealthDamaged
    - PlacedItem.cs               - ApplyItemEffect()
    - ItemStack.cs
                |                          ^
    ServiceLocator.Register         | EventBus
    Publish:                       |
    - InventoryChanged             |
    - InventoryViewChanged         |
    - EquipmentChanged             |
    - WeaponEquippedEvent          |
    - WeaponUnequippedEvent        |
                |                          |
                v                          |
          ItemUsageSystem  ← Subscribe: ItemUsedEvent
          → Inventory.RemoveItem()
          → Survival.ApplyItemEffect()
                ^
                | EventBus.Publish(ItemUsedEvent)
                |
          QuickItemBar UI ← Subscribe: InventoryChanged
          Publish: ItemUsedEvent

    ┌─────── 关联系统（消费者）───────────┐
    │ WeaponHolder     ← WeaponEquipped/Unequipped
    │ WeaponSwitcher   → WeaponSlotChanged
    │ WeaponShooting   ← 读 Equipped[slot]
    │ PlayerCharacter  → 力量影响负重
    │ InventoryUI      ← InventoryViewChanged
    │ DragDropManager  ← 操作 Inventory/Container
    │ CraftingSystem   → RemoveItem/AddItem
    │ SaveLoadManager  → GetSaveData/RestoreFromSave
    │ AIBotInventory   → 复用 Inventory 组件
    │ WorldContainer   → 拾取/容器交互
    └─────────────────────────────────────┘
```

**关键关联方式**：
- **ServiceLocator**：`Inventory` 自身注册（`Awake`），`InventoryUI` / `QuickItemBar` / `ItemUsageSystem` / `SaveLoadManager` 通过 `ServiceLocator.Get<Inventory>()` 获取
- **EventBus**：所有状态变更通过事件广播，订阅者解耦
- **直接引用**：`PlayerCharacter` 组件（同 GameObject 上的 `GetComponent`）
- **ScriptableObject 引用**：`ItemData` 作为资产文件，各处通过 `AssetDatabase.LoadAssetAtPath` 或运行时 `ItemCatalog` 查找

### EventBus 事件

#### 物品系统发布的事件

| 事件 | 发布者 | 触发时机 | 订阅者 |
|---|---|---|---|
| `InventoryChanged` | Inventory | 添加/移除/装备/卸下/超载警告/槽满 | QuickItemBar, InventoryUI |
| `InventoryViewChangedEvent` | Inventory.PublishView() | 任何背包变化后 | InventoryUI（完整面板刷新） |
| `EquipmentChangedEvent` | Inventory | 装备/卸下后（携带 totalArmor + totalWarmth） | SurvivalSystem（更新护甲/保暖缓存） |
| `WeaponEquippedEvent` | Inventory.EquipItem() | 武器装备到武器槽 | WeaponHolder（生成武器模型） |
| `WeaponUnequippedEvent` | Inventory.UnequipSlot() | 武器从武器槽卸下 | WeaponHolder（销毁武器模型） |
| `ItemUsedEvent` | QuickItemBar | 右键使用物品后（携带 ItemData + count） | ItemUsageSystem（扣物品+加效果） |

#### 物品系统订阅的事件

| 事件 | 订阅者 | 用途 |
|---|---|---|
| `ItemUsedEvent` | ItemUsageSystem | 扣背包物品 + 应用生存效果 |
| `InventoryChanged` | QuickItemBar | 刷新快捷栏显示 |

#### 物品系统内联调用（非 EventBus，直接方法调用）

| 调用 | 调用者 → 被调用者 | 用途 |
|---|---|---|
| `Inventory.RemoveItem()` | ItemUsageSystem, QuickItemBar, CraftingSystem | 消耗/使用/合成扣除物品 |
| `Inventory.AddItem()` | CraftingSystem, WorldContainer | 合成产物/拾取加入背包 |
| `Inventory.EquipItem()` | InventoryUI / DragDropManager | UI 拖放装备 |
| `Inventory.GetSaveData()` | SaveLoadManager | 存档采集 |
| `Inventory.RestoreFromSave()` | SaveLoadManager | 读档恢复 |
| `Inventory.CountItemByName()` | WeaponShooting, ProductionDevice | 弹药/燃料存量查询 |
| `Inventory.RemoveItemByName()` | WeaponShooting | 换弹扣弹药 |
