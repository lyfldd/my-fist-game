────────────────────────────────────────
一、数组的声明与初始化
────────────────────────────────────────

// 完整写法
变量类型[] 数组名 = new 变量类型[长度] { 元素1, 元素2, ... };

int[] a = new int[5] { 1, 2, 3, 4, 5 };

// 省略长度（编译器自动推断）
int[] b = new int[] { 1, 2, 3 };

// 最简写法
int[] c = { 1, 2, 3, 4, 5 };

// 只声明长度，不赋初值（元素自动填充默认值）
int[] d = new int[5];
// 数值类型默认值为 0，bool 为 false，string 为 null，引用类型为 null

────────────────────────────────────────
二、数组的基本操作
────────────────────────────────────────

int[] a = new int[5] { 1, 2, 3, 4, 5 };

// 获取长度
Console.WriteLine(a.Length);    // 输出 5

// 访问元素（下标从 0 开始）
Console.WriteLine(a[0]);   // 1
Console.WriteLine(a[4]);   // 5

// 修改元素
a[1] = 90;   // 第二个元素改为 90

// 遍历数组（for 循环）
for (int i = 0; i < a.Length; i++)
{
    Console.WriteLine(a[i]);
}

// 遍历数组（foreach，更简洁）
foreach (int num in a)
{
    Console.WriteLine(num);
}

────────────────────────────────────────
三、增加数组元素（搬家法）
────────────────────────────────────────

// 数组初始化后长度固定，不能直接增减
// 方法：新建更大的数组，把旧数据复制过去

int[] a = new int[5] { 1, 2, 3, 4, 5 };
int[] b = new int[6];   // 新数组多一个位置
for (int i = 0; i < a.Length; i++)
{
    b[i] = a[i];
}
b[5] = 6;   // 填入新元素
a = b;      // a 指向新数组（原来的数组等待垃圾回收）

────────────────────────────────────────
四、删除数组元素
────────────────────────────────────────

// 删除末尾元素
int[] a = new int[5] { 1, 2, 3, 4, 5 };
int[] b = new int[4];
for (int i = 0; i < b.Length; i++)
{
    b[i] = a[i];
}
a = b;

// 删除中间某个元素（例如下标为 1 的元素）
int[] a = new int[5] { 1, 2, 3, 4, 5 };
int[] b = new int[4];
int j = 0;
for (int i = 0; i < a.Length; i++)
{
    if (i == 1)
        continue;     // 跳过下标 1
    b[j] = a[i];
    j++;
}
a = b;

七、二维数组的声明与使用
────────────────────────────────────────

// 声明（行 × 列）
int[,] arr2D = new int[3, 3];

// 声明并初始化
int[,] c = new int[3, 3]
{
    { 1, 2, 3 },    // 第 0 行
    { 4, 5, 6 },    // 第 1 行
    { 7, 8, 9 }     // 第 2 行
};

// 获取行数和列数
int rows = c.GetLength(0);     // 行数 = 3
int cols = c.GetLength(1);     // 列数 = 3

// 访问元素
Console.WriteLine(c[0, 1]);   // 第 0 行第 1 列 = 2

// 遍历二维数组
for (int i = 0; i < c.GetLength(0); i++)
{
    for (int j = 0; j < c.GetLength(1); j++)
    {
        Console.Write(c[i, j] + " ");
    }
    Console.WriteLine();
}

八、交错数组（数组的数组）
────────────────────────────────────────

// 交错数组：每行的列数可以不同（不规则）
// 普通二维数组 int[,]  vs  交错数组 int[][]

// 声明：变量类型[][]
int[][] a = new int[3][];
a[0] = new int[] { 1, 2 };           // 第 0 行 2 列
a[1] = new int[] { 7, 8, 9 };        // 第 1 行 3 列
a[2] = new int[] { 0 };              // 第 2 行 1 列

// 获取行数
int rows = a.GetLength(0);           // 或 a.Length

// 获取某行的列数
int cols = a[0].Length;

// 访问元素
Console.WriteLine(a[0][0]);         // 第 0 行第 0 列 = 1

// 遍历交错数组
for (int i = 0; i < a.Length; i++)
{
    for (int j = 0; j < a[i].Length; j++)
    {
        Console.Write(a[i][j] + " ");
    }
    Console.WriteLine();
}

// 存储二维数组的交错数组（高级用法）
int[][,] twoDArrayCollection = new int[3][,]
{
    new int[2, 2] { { 1, 2 }, { 3, 4 } },
    new int[3, 3] { { 5, 6, 7 }, { 8, 9, 10 }, { 11, 12, 13 } },
    new int[2, 3] { { 14, 15, 16 }, { 17, 18, 19 } }
};

────────────────────────────────────────
九、枚举的定义
────────────────────────────────────────

// 枚举：自定义值类型，是常量的集合
// 核心作用：给一组固定的整数常量赋予语义化名称

// 语法
enum E_枚举名
{
    成员1,        // 默认值 0
    成员2,        // 默认值 1
    成员3 = 10,   // 手动指定值为 10
    成员4,        // 紧接着是 11
}

// 声明位置：
//   ✅ namespace 语句块中
//   ✅ class 语句块中
//   ✅ struct 语句块中
//   ❌ 函数语句块中（不能在函数内声明枚举）

// 示例
enum Monster
{
    normal,   // = 0
    boss,     // = 1
}

────────────────────────────────────────
十、枚举的使用
────────────────────────────────────────

// 声明枚举变量
Monster monster = Monster.boss;

// 配合 if 判断
if (monster == Monster.boss)
{
    Console.WriteLine("这是 boss");
}

// 配合 switch（枚举 + switch 是绝配）
switch (monster)
{
    case Monster.normal:
        Console.WriteLine("普通怪物");
        break;
    case Monster.boss:
        Console.WriteLine("Boss！");
        break;
    default:
        break;
}

────────────────────────────────────────
十一、枚举与类型的相互转换
────────────────────────────────────────

Monster monster = Monster.boss;

// 枚举 → int
int i = (int)monster;
Console.WriteLine(i);           // 输出 1

// 枚举 → string
string str = monster.ToString();
Console.WriteLine(str);         // 输出 "boss"

// string → 枚举（用 Enum.Parse）
monster = (Monster)Enum.Parse(typeof(Monster), "boss");

// int → 枚举（直接强转）
monster = (Monster)0;           // monster = Monster.normal

────────────────────────────────────────
十二、枚举的实际用途
────────────────────────────────────────

// 游戏开发中，对象常有多种状态，枚举是最佳表达方式

enum E_PlayerState
{
    Idle,       // 待机
    Walk,       // 走
    Run,        // 跑
    Attack,     // 攻击
    Dead,       // 死亡
}

E_PlayerState state = E_PlayerState.Idle;
// 比用 int 数字来标识状态更清晰、更不容易出错
