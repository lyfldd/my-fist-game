────────────────────────────────────────
一、函数的基本概念
────────────────────────────────────────

函数（方法）：具有名字的代码块
- 可以通过函数名来调用这段代码，避免重复编写
- 声明位置：class 语句块中 或 struct 语句块中
- 不能在函数内部嵌套声明函数

────────────────────────────────────────
二、函数的声明语法
────────────────────────────────────────

static 返回类型 函数名(参数类型 参数名1, 参数类型 参数名2)
{
    // 函数体（代码块）
    return 返回值;   // 有返回类型才写 return
}

// 说明：
// static：静态修饰符（在 Main 函数所在的 Program 类里，其他函数也需要 static 才能被 Main 调用）
// 返回类型：void 表示无返回值；其他类型见下文
// 参数：可以有 0 个或多个，格式是"类型 名字"，多个参数用逗号隔开

────────────────────────────────────────
三、四种常见函数类型
────────────────────────────────────────

① 无参无返回值
static void Greet()
{
    Console.WriteLine("你好世界");
}
// 调用
Greet();

② 有参数无返回值
static void PrintName(string name)
{
    Console.WriteLine("你的名字是 {0}", name);
}
// 调用（传常量、变量、表达式都可以，类型要匹配）
PrintName("张三");

③ 无参数有返回值
static string GetName()
{
    return "张三";
}
// 调用：返回值必须"用掉"（赋值给变量或直接拿来用）
string name = GetName();
Console.WriteLine(GetName());

④ 有参数有返回值
static int Add(int a, int b)
{
    return a + b;
}
// 调用
int result = Add(3, 5);    // result = 8

────────────────────────────────────────
五、返回多个值的两种方式
────────────────────────────────────────

// 方式 1：返回数组（把多个结果打包到数组里）
static int[] SumAndDiff(int a, int b)
{
    int sum = a + b;
    int diff = a - b;
    return new int[] { sum, diff };
}
// 调用
int[] result = SumAndDiff(10, 3);
Console.WriteLine("和={0}, 差={1}", result[0], result[1]);

// 方式 2：用 out 参数（见 ref和out 章节）
static void SumAndDiff2(int a, int b, out int sum, out int diff)
{
    sum = a + b;
    diff = a - b;
}
// 调用
SumAndDiff2(10, 3, out int s, out int d);
Console.WriteLine("和={0}, 差={1}", s, d);

────────────────────────────────────────
六、return 的两个作用
────────────────────────────────────────

// 作用 1：返回值（有返回类型时）
static int GetMax(int a, int b)
{
    if (a > b)
        return a;
    return b;
}

// 作用 2：提前结束函数（即使没有返回值）
static void PrintPositive(int n)
{
    if (n <= 0)
    {
        Console.WriteLine("不是正数，退出");
        return;     // 直接结束函数，后面的代码不再执行
    }
    Console.WriteLine("正数：" + n);
}

七、局部变量作用域（补充）
────────────────────────────────────────

// 函数内部声明的变量是"局部变量"，只在该函数内有效
// 两个函数里可以有同名变量，互不影响

static void A()
{
    int x = 10;   // 这个 x 只在 A() 里
}

static void B()
{
    int x = 20;   // 这个 x 只在 B() 里，和 A() 的 x 无关
}

// for/if/while 代码块内声明的变量，作用域只在该代码块内
for (int i = 0; i < 5; i++)
{
    int temp = i * 2;   // temp 只在这个 for 循环的大括号内有效
}
// 这里访问 temp 会报错

// C# 约定函数名用 PascalCase（每个单词首字母大写）
