List<T> —— 遍历
=============================================

一、三种遍历方式

// 方式 1：foreach（最推荐）
List<int> list = new List<int> { 1, 2, 3, 4, 5 };

foreach (int item in list)
{
    Console.WriteLine(item);
}

// 方式 2：for 循环
for (int i = 0; i < list.Count; i++)
{
    Console.WriteLine(list[i]);
}

// 方式 3：ForEach（List 专属方法）
list.ForEach(item =>
{
    Console.WriteLine(item);
});


二、for 循环的优势：边遍历边操作索引

// 正序遍历并删除所有偶数
for (int i = 0; i < list.Count; i++)
{
    if (list[i] % 2 == 0)
    {
        list.RemoveAt(i);
        i--;  // 删除后索引要回退，否则会跳过下一个元素
    }
}

// 正序遍历并修改元素
for (int i = 0; i < list.Count; i++)
{
    list[i] *= 2;  // 每个元素翻倍
}


三、倒序遍历（适合删除操作）

List<int> list = new List<int> { 1, 2, 3, 4, 5 };

// 倒序遍历删除所有大于 3 的元素
for (int i = list.Count - 1; i >= 0; i--)
{
    if (list[i] > 3)
    {
        list.RemoveAt(i);
    }
}
// 删除时不会影响未访问的元素，逻辑更安全


四、ForEach 的限制

⚠️ ForEach 不允许在回调内修改列表（添加/删除/清空）

list.ForEach(item =>
{
    if (item == 2)
        list.Remove(item);  // ❌ 抛 InvalidOperationException
});

// 如果需要边遍历边删，用 for 倒序遍历


五、遍历时获取索引

// 方式 1：for 循环（自带 i）
for (int i = 0; i < list.Count; i++)
{
    Console.WriteLine($"索引 {i}: {list[i]}");
}

// 方式 2：foreach + 手动计数器
int index = 0;
foreach (var item in list)
{
    Console.WriteLine($"索引 {index}: {item}");
    index++;
}


六、Count vs Capacity（补充说明）

  Count     → 实际元素数量（我们用的是这个）
  Capacity  → 内部数组容量

// 示例
List<int> list = new List<int>(100);  // Capacity = 100，但 Count = 0
list.Add(1);                          // Count = 1，Capacity 仍为 100
list.Add(2);                          // Count = 2
list.Clear();                         // Count = 0，Capacity 仍为 100

// 压缩多余容量
list.TrimExcess();  // Capacity → Count（节省内存，但小列表没必要调用）
